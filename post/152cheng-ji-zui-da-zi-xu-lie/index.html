class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int imax=1,imin=1,ans=INT_MIN;

        for(int i:nums)
        {
            if(i<0)
                swap(imin,imax);

            imax=max(imax*i,i);
            imin=min(imin*i,i);
        
            ans=max(ans,imax);
        }
        return ans;
    }
};
